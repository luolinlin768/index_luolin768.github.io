  • Kruskal算法（稀疏图最小生成树）

先把边按照权值进行排序，用贪心的思想优先选取权值较小的边，并依次连接，若出现环则跳过此边（用并查集来判断是否存在环）继续搜，直到已经使用的边的数量比总点数少一即可。

//前面已经弄好并查集了，注意初始化
struct edge{
    int s, e, v, next;
    bool operator < (const edge &x)const{
        return v > x.v;
    }
}eg[maxm];
int head[maxn], cnt;
priority_queue <edge> q;
int kruskal(){
    int ans = 0;
    for(int i = 1; i <= cnt; ++i)q.push(eg[i]);//每一条边丢入优先队列
    while(!q.empty()){
        edge t = q.top();
        q.pop();
        if(find(t.s) ！= find(t.e)){
            ans += t.v;
            Union(t.s, t.e);
        }
    }
    return ans;
}
  • Prim算法（稠密图最小生成树）

将任意节点作为根，再找出与之相邻的所有边（用一遍循环即可），再将新节点更新并以此节点作为根继续搜，维护一个数组：dis[]，作用为已用点到未用点的最短距离。

struct node {
    int dis;
    int pos;//当前访问结点
    int from;//这条边从哪来
    bool operator < (const node& x)const {
        return x.dis<dis;
    }
};
//升序队列，小根堆
std::priority_queue <node> q;
int Prim(){
    for(int i = 2; i <= n; ++i)dis[i] = INF;//初始化
    q.push({0, 1, 1});//(1,1)的距离为0
    while(!q.empty() && cnt < n){
        node now = q.top();
        int x = now.pos, d = now.dis;//x为当前访问结点
        q.pop();
        if(vis[x])continue;
        vis[x] = 1;
        ++cnt;
        ans += d;
        for(int i = head[x]; i; i = e[i].next){
            if(e[i].weight < dis[e[i].to]){
                dis[e[i].to] = e[i].weight;
                q.push({dis[e[i].to], e[i].to, x});
            }
        }
    }
    return ans;
}

void dij(int s){
    memset(dis, 0x7f, sizeof(dis));
    dis[s] = 0;
    q.push(make_pair(dis[s], s));
    while(!q.empty()){
        int t = q.top().second;
        q.pop();
        if(vis[t])continue;
        vis[t]=1;
        for(int i = 0; i < p[t].size(); ++i){
            if(dis[p[t][i].first] > dis[t]+p[t][i].second){
                dis[p[t][i].first] = dis[t]+p[t][i].second;
                q.push(make_pair(dis[p[t][i].first], p[t][i].first));
            }
        }
    }
}

int lg[MAXN];

int fa[MAXN][40];

void getlog(){
    for(int i = 1 ; i <= n ; ++i)lg[i] = lg[i >> 1] + 1;

void dfs(int id, int fath){
    dep[id] = dep[fath] + 1;
    fa[id][0] = fath;
    for(int i = 1; i <= lg[dep[id]]; ++i)fa[id][i] = fa[fa[id][i -  1]][i - 1];
    for(int i = 0; i < p[id].size(); ++i)if(p[id][i] != fath)dfs(p[id][i], id);
}

int lca(int a, int b){
    if(dep[a] < dep[b])swap(a, b);
    while(dep[a] != dep[b])a = fa[a][lg[dep[a] - dep[b]] - 1];
    if(a == b)return a;//格外注意 必须有
    for(int i = lg[dep[a]]; i >= 0; --i){
        if(fa[a][i] != fa[b][i]){
            a = fa[a][i];
            b = fa[b][i];
        }
    }
    return fa[a][0];
}

  • 最长公共子列（lcs）

关于为什么可以转化成LIS问题，这里提供一个解释。

A:3 2 1 4 5

B:1 2 3 4 5

我们不妨给它们重新标个号：把3标成a,把2标成b，把1标成c……于是变成：

A: a b c d e

B: c b a d e

这样标号之后，LCS长度显然不会改变。但是出现了一个性质：两个序列的子序列，一定是A的子序列。而A本身就是单调递增的。

因此这个子序列是单调递增的。

换句话说，只要这个子序列在B中单调递增，它就是A的子序列。

哪个最长呢？当然是B的LIS最长。自此完成转化。

#include<bits/stdc++.h>
using namespace std;
int a[100005];
int b[100005];
int n;
int lis(){
    int len = n;
    int s[200005],t = 0;
    s[++t] = b[1];
    for(int i = 2; i <= len; ++i){
        if(b[i] > s[t])s[++t] = b[i];
        else{
            int temp = lower_bound(s,s+t+1,b[i])-s;
            s[temp] = b[i];
        }
    }
    return t;
}

int main(){
    n = read();
    for(int i=1; i<=n; ++i){
        int t = read();
        a[t] = i;
    }
    for(int i=1; i<=n; ++i){
        int t = read();
        b[i] = a[t];
    }
    cout << lis();
    return 0;
}
  • 加法乘法线段树

【输入格式】

第一行包含三个整数 n,q,mn,q,mn,q,m，分别表示该数列数字的个数、操作的总个数和模数。

第二行包含 nnn 个用空格分隔的整数，其中第 iii 个数字表示数列第 iii 项的初始值。

接下来 qqq 行每行包含若干个整数，表示一个操作，具体如下：

操作 111： 格式：1 x y k 含义：将区间 [x,y][x,y][x,y] 内每个数乘上 kkk

操作 222： 格式：2 x y k 含义：将区间 [x,y][x,y][x,y] 内每个数加上 kkk

操作 333： 格式：3 x y 含义：输出区间 [x,y][x,y][x,y] 内每个数的和对 mmm 取模所得的结果

int n, m, mod;
ll tree[100000 * 4 + 5];//注意四倍空间
ll add[100000 * 4 + 5];
ll pos[100000 * 4 + 5];
int arr[100000 + 5];
void build(int l, int r, int id){
    pos[id] = 1;
    if(l == r){
        tree[id] = arr[l] % mod;
        return;
    }
    int mid = (l + r) >> 1;
    build(l, mid, id * 2);
    build(mid + 1, r, id * 2 + 1);
    tree[id] = (tree[id * 2] + tree[id * 2 + 1]) % mod;
}
void lazy(int id, int l, int r){
    int mid = (l + r) >> 1;
    tree[id * 2] = (tree[id * 2] * pos[id] + add[id] * (mid - l + 1)) % mod;
    tree[id * 2 + 1] = (tree[id * 2 + 1] * pos[id] + add[id] * (r - mid)) % mod;
    pos[id * 2] = (pos[id * 2] * pos[id]) % mod;
    pos[id * 2 + 1] = (pos[id * 2 + 1] * pos[id]) % mod;
    add[id * 2] = (add[id * 2] * pos[id] + add[id]) % mod;
    add[id * 2 + 1] = (add[id * 2 + 1] * pos[id] + add[id]) % mod;
    pos[id] = 1;
    add[id] = 0;
}

void ad(ll v, int l, int r, int ml, int mr, int id){
    if(ml <= l && r <= mr){
        tree[id] = tree[id] + (v * (r - l + 1)) % mod;
        add[id] = (add[id] + v) % mod;
        return;
    }
    if(l > mr || r < ml)return;
    lazy(id, l, r);
    int mid = (l + r) >> 1;
    ad(v, l, mid, ml, mr, id * 2);
    ad(v, mid + 1, r, ml, mr, id * 2 + 1);
    tree[id] = (tree[id * 2] + tree[id * 2 + 1]) % mod;
}

void po(ll v, int l, int r, int ml, int mr, int id){
    if(ml <= l && r <= mr){
        tree[id] = (tree[id] * v) % mod;
        pos[id] = (pos[id] * v) % mod;
        add[id] = (add[id] * v) % mod;
        return;
    }
    if(l > mr || r < ml)return;
    lazy(id, l, r);
    int mid = (l + r) >> 1;
    po(v, l, mid, ml, mr, id * 2);
    po(v, mid + 1, r, ml, mr, id * 2 + 1);
    tree[id] = (tree[id * 2] + tree[id * 2 + 1]) % mod;
}

ll query(int l, int r, int ml, int mr, int id){
    if(ml <= l && r <= mr)
        return tree[id];
    if(l > mr || r < ml)return 0;
    lazy(id, l, r);
    int mid = (l + r) >> 1;
    return (query(l, mid, ml, mr, id * 2) + query(mid + 1, r, ml, mr, id * 2 + 1)) % mod;
}

int main(){
    n = rd(), m = rd(), mod = rd();
    for(int i = 1; i <= n; ++i)arr[i] = rd();
    build(1, n, 1);
    while(m--){
        int sign = rd();
        if(sign == 1){
            ll x = rd(), y = rd(), k = rd();
            po(k, 1, n, x, y, 1);
        }
        if(sign == 2){
            ll x = rd(), y = rd(), k = rd();
            ad(k, 1, n, x, y, 1);
        }
        if(sign == 3){
            ll x = rd(), y = rd();
            printf("%lld\n", query(1, n, x, y, 1));
        }
    }
    return 0;
}

#include<bits/stdc++.h>
using i64 = long long;

struct node{
    int cnt = 0;
    node *nxt[62] = {};
};

int hash(char ch){
    if(ch >= '0' && ch <= '9')return ch - '0' + 52;
    if(ch >= 'a' && ch <= 'z')return ch - 'a';
    return ch - 'A' + 26;
}

void insert(int id, std::string s, node *root){
    ++root -> cnt;
    if(id == s.length())return;
    int h = hash(s[id]);
    if(root -> nxt[h] == NULL)root -> nxt[h] = new node;
    insert(id + 1, s, root -> nxt[h]);
}

int query(int id, std::string s, node *root){
    if(id == s.length())return root -> cnt;
    int h = hash(s[id]);
    if(root -> nxt[h] == NULL)return 0;
    return query(id + 1, s, root -> nxt[h]);
}

void del(node *root){
    for(int i = 0; i < 62; ++i)if(root -> nxt[i] != NULL)del(root -> nxt[i]);
    delete root;
}

int main(){
    std::ios::sync_with_stdio(0);
    std::cin.tie(0);

    int t;
    std::cin >> t;
    while(t--){
        node *root = new node;
        int n, q;
        std::cin >> n >> q;
        while(n--){
            std::string s;
            std::cin >> s;
            insert(0, s, root);
        }
        while(q--){
            std::string s;
            std::cin >> s;
            std::cout << query(0, s, root) << '\n';
        }
        del(root);
    }

    return 0;
}
